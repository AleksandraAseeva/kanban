```
// tasks.js (Vuex module)
const tasksModule = {
  namespaced: true,
  state: { tasks: [] },
  mutations: {
    ADD_TASK(state, task) {
      state.tasks.push(task);
      this.commit('tasks/saveTasks');
    },
    saveTasks(state) {
      Cookies.set('tasks', JSON.stringify(state.tasks), { expires: 7 });
    },
    loadTasks(state) {
      const tasks = Cookies.get('tasks');
      if (tasks) state.tasks = JSON.parse(tasks);
    }
  }
};
```
